function startRec(){

  if(!camStream){showToast('Caméra non disponible');return;}

  chunks=[];pendBlob=null;

  const mt=MediaRecorder.isTypeSupported('video/webm;codecs=vp9')?'video/webm;codecs=vp9':

           MediaRecorder.isTypeSupported('video/webm')?'video/webm':'video/mp4';

  mr=new MediaRecorder(camStream,{mimeType:mt});

  mr.ondataavailable=e=>{if(e.data.size>0)chunks.push(e.data);};

  mr.onstop=()=>{

    pendBlob=new Blob(chunks,{type:mt});

    document.getElementById('play-vid').src=URL.createObjectURL(pendBlob);

    document.getElementById('play-wrap').classList.add('show');

    document.getElementById('preview-wrap').style.display='none';

    document.getElementById('row1').style.display='none';

    document.getElementById('row2').style.display='flex';

    document.getElementById('row2').style.flexDirection='column';

    document.getElementById('cam-status').textContent='👆 Vérifiez puis sauvegardez sur Supabase';

  };

  mr.start(100);

  document.getElementById('btn-start').style.display='none';

  document.getElementById('btn-stop').style.display='block';

  document.getElementById('rec-ind').classList.add('show');

  document.getElementById('rec-timer').classList.add('show');

  recSec=0;

  timerIv=setInterval(()=>{recSec++;const m=Math.floor(recSec/60),s=recSec%60;document.getElementById('rec-timer').textContent=`${m}:${s.toString().padStart(2,'0')}`;},1000);

}

function stopRec(){

  if(mr&&mr.state!=='inactive')mr.stop();

  clearInterval(timerIv);

  document.getElementById('rec-ind').classList.remove('show');

  document.getElementById('rec-timer').classList.remove('show');

  document.getElementById('btn-stop').style.display='none';

}

async function saveAndUpload(){

  if(!pendBlob){showToast('Enregistre d\'abord une vidéo');return;}

  const btn=document.getElementById('btn-upload');

  btn.disabled=true;

  document.getElementById('upload-bar-wrap').classList.add('show');

  document.getElementById('upload-status').classList.add('show');

  document.getElementById('upload-status').textContent='⬆️ Upload en cours…';


  try {

    const ext = pendBlob.type.includes('mp4') ? 'mp4' : 'webm';

    const path = `${groomSessionId}/q_${gIdx}.${ext}`;


    // Fake progress while uploading

    let prog=0;

    const fakeIv=setInterval(()=>{prog=Math.min(prog+5,90);document.getElementById('upload-bar').style.width=prog+'%';},200);


    const {error:upErr} = await sb.storage.from(BUCKET).upload(path, pendBlob, {upsert:true, contentType:pendBlob.type});

    clearInterval(fakeIv);

    if(upErr) throw upErr;


    document.getElementById('upload-bar').style.width='100%';


    const {data:urlData} = sb.storage.from(BUCKET).getPublicUrl(path);

    const publicUrl = urlData.publicUrl;


    // Update DB

    const qId = Qs[gIdx].id;

    if(qId){

      await sb.from('evjf_questions').update({video_path:path, video_url:publicUrl, recorded_at: new Date().toISOString()}).eq('id', qId);

    }

    Qs[gIdx].video_path = path;

    Qs[gIdx].video_url  = publicUrl;


    document.getElementById('upload-status').textContent='✅ Vidéo sauvegardée !';

    showToast('Vidéo uploadée sur Supabase ✓');

    renderGroomQ();


    // Auto-advance to next unanswered

    const nx=Qs.findIndex((q,i)=>i>gIdx&&!q.video_url);

    if(nx!==-1)setTimeout(()=>{gIdx=nx;renderGroomQ();},800);

  } catch(e){

    document.getElementById('upload-status').textContent='❌ Erreur upload : ' + e.message;

    showToast('Erreur : ' + e.message);

  } finally {

    btn.disabled=false;

  }

}
